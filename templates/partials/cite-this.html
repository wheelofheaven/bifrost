{# Page "Cite this page" widget (Decision 15, H1.3). Computes citation fields from page metadata and defers rendering to macros/cite.html. Year falls back date -> updated -> editorial_pass -> n.d. #} {%- set _author = page.extra.author | default(value=config.extra.author) | default(value=config.title) -%} {%- if page.date %}{% set_global _cite_year = page.date | date(format="%Y") %} {%- elif page.updated %}{% set_global _cite_year = page.updated | date(format="%Y") %} {%- elif page.extra.editorial_pass %}{% set_global _cite_year = page.extra.editorial_pass | truncate(length=4, end="") %} {%- else %}{% set_global _cite_year = "n.d." %}{%- endif -%} {{ cite::widget( label=trans(key="citeThisPage", lang=detected_lang), title=page.title, url=page.permalink, slug=page.slug, year=_cite_year, site=config.title, author=_author, is_book=false, entry_type="misc", note="CC0-1.0 public domain") }}
