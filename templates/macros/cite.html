{# Shared citation widget (Decision 15, H1.3). Renders the .cite
block with APA / MLA / Chicago / BibTeX. Two callers: - partials/cite-this.html → page citation (is_book=false, title-first, @misc) - partials/cite-book.html → book citation (is_book=true, author-first, @book) The copy-to-clipboard behaviour lives in static/js/cite-copy.js (core bundle), so this macro emits no inline script. #} {% macro widget(label, title, url, slug, year, site, author="", is_book=false, entry_type="misc", note="CC0-1.0 public domain") %} {%- set lb = "{" -%}{%- set rb = "}" -%} {# Skip the period after a title that already ends in terminal punctuation. #} {%- set_global _dot = "." -%} {%- if title is ending_with("?") or title is ending_with("!") or title is ending_with(".") -%}{%- set_global _dot = "" -%}{%- endif -%}
{{ label }}
APA
{% if is_book and author %}{{ author }}. ({{ year }}). {{ title }}{{ _dot }}{% else %}{{ title }}{{ _dot }} ({{ year }}).{% endif %} {{ site }}. {{ url | safe }}
MLA
{% if is_book and author %}{{ author }}. {{ title }}{{ _dot }} {{ year }}. {{ site }}, {{ url | safe }}.{% else %}"{{ title }}{{ _dot }}" {{ site }}, {{ year }}, {{ url | safe }}.{% endif %}
Chicago
{% if is_book and author %}{{ author }}. {{ title }}{{ _dot }} {{ year }}. {{ site }}. {{ url | safe }}.{% else %}"{{ title }}{{ _dot }}" {{ site }}, {{ year }}. {{ url | safe }}.{% endif %}

{% if author %}  author       = {{ lb }}{% if not is_book %}{{ lb }}{% endif %}{{ author }}{% if not is_book %}{{ rb }}{% endif %}{{ rb }},
{% endif %}  title        = {{ lb }}{{ title }}{{ rb }},
  year         = {{ lb }}{{ year }}{{ rb }},
  howpublished = {{ lb }}\url{{ lb }}{{ url }}{{ rb }}{{ rb }},
  note         = {{ lb }}{{ note }}{{ rb }}
{{ rb }}
{% endmacro widget %}
